def sales_analysis():
    conn = sqlite3.connect('retail_analysis.db')
    
    monthly_sales = pd.read_sql_query("""
        SELECT 
            CASE 
                WHEN substr(Date, 4, 2) = '01' THEN 'January'
                WHEN substr(Date, 4, 2) = '02' THEN 'February'
                WHEN substr(Date, 4, 2) = '03' THEN 'March'
                WHEN substr(Date, 4, 2) = '04' THEN 'April'
                WHEN substr(Date, 4, 2) = '05' THEN 'May'
                WHEN substr(Date, 4, 2) = '06' THEN 'June'
                WHEN substr(Date, 4, 2) = '07' THEN 'July'
                WHEN substr(Date, 4, 2) = '08' THEN 'August'
                WHEN substr(Date, 4, 2) = '09' THEN 'September'
                WHEN substr(Date, 4, 2) = '10' THEN 'October'
                WHEN substr(Date, 4, 2) = '11' THEN 'November'
                WHEN substr(Date, 4, 2) = '12' THEN 'December'
            END as Month,
            substr(Date, 7, 4) as Year,
            ROUND(SUM("Total Amount"), 2) as Total_Sales
        FROM sales_transactions
        GROUP BY substr(Date, 4, 2), substr(Date, 7, 4)
        ORDER BY Year, substr(Date, 4, 2)
    """, conn)
    
    print("\nMonthly Sales Data:")
    print(monthly_sales)
    
    # Plotting
    plt.figure(figsize=(12, 6))
    plt.plot(range(len(monthly_sales)), monthly_sales['Total_Sales'], marker='o')
    plt.title('Monthly Sales Trend')
    plt.xlabel('Month')
    plt.ylabel('Total Sales (₹)')
    plt.xticks(range(len(monthly_sales)), monthly_sales['Month'] + ' ' + monthly_sales['Year'], rotation=45)
    plt.grid(True)
    plt.tight_layout()
    plt.show()
    
    conn.close()
    return monthly_sales

# Run the analysis
sales_data = sales_analysis()
