def advanced_retail_analysis():
    conn = sqlite3.connect('retail_analysis.db')
    
    # Complex analysis combining product categories, stores, and time
    analysis = pd.read_sql_query("""
        WITH monthly_category_sales AS (
            SELECT 
                strftime('%m', s.Date) as Month,
                st.Location,
                p."Product Name",
                p.Category,
                COUNT(DISTINCT s."Transaction ID") as Num_Transactions,
                SUM(s.Quantity) as Units_Sold,
                ROUND(SUM(s."Total Amount"), 2) as Revenue,
                ROUND(SUM(s."Total Amount")/SUM(s.Quantity), 2) as Avg_Unit_Price
            FROM sales_transactions s
            JOIN products p ON s."Product ID" = p."Product ID"
            JOIN stores st ON s."Store ID" = st."Store ID"
            GROUP BY Month, st.Location, p.Category
        )
        SELECT 
            Month,
            Location,
            Category,
            Num_Transactions,
            Units_Sold,
            Revenue,
            Avg_Unit_Price,
            ROUND(Revenue * 100.0 / SUM(Revenue) OVER (PARTITION BY Month), 2) as Monthly_Revenue_Share
        FROM monthly_category_sales
        ORDER BY Month, Revenue DESC
    """, conn)
    
    # Convert month numbers to names
    month_names = {
        '01': 'January', '02': 'February', '03': 'March', '04': 'April',
        '05': 'May', '06': 'June', '07': 'July', '08': 'August',
        '09': 'September', '10': 'October', '11': 'November', '12': 'December'
    }
    analysis['Month'] = analysis['Month'].map(month_names)

    # Print detailed analysis
    print("\nCross-Store Category Performance Analysis:")
    print("=" * 80)
    
    # Summary by Location
    location_summary = analysis.groupby('Location')['Revenue'].sum().sort_values(ascending=False)
    print("\nStore Location Performance:")
    for loc in location_summary.index:
        print(f"\n{loc}:")
        print(f"Total Revenue: ₹{location_summary[loc]:,.2f}")
        print(f"Top Categories:")
        top_cats = analysis[analysis['Location'] == loc].groupby('Category')['Revenue'].sum().nlargest(3)
        for cat, rev in top_cats.items():
            print(f"- {cat}: ₹{rev:,.2f}")
    
    # Visualizations
    plt.figure(figsize=(15, 10))
    
    # Plot 1: Category Performance by Store
    plt.subplot(2, 1, 1)
    pivot_data = analysis.pivot_table(
        values='Revenue', 
        index='Location', 
        columns='Category', 
        aggfunc='sum'
    )
    pivot_data.plot(kind='bar', stacked=True)
    plt.title('Category Revenue Distribution by Store')
    plt.xlabel('Store Location')
    plt.ylabel('Revenue (₹)')
    plt.legend(title='Category', bbox_to_anchor=(1.05, 1), loc='upper left')
    plt.xticks(rotation=45)
    
    # Plot 2: Monthly Trend of Top Categories
    plt.subplot(2, 1, 2)
    top_categories = analysis.groupby('Category')['Revenue'].sum().nlargest(3).index
    monthly_cat_trend = analysis[analysis['Category'].isin(top_categories)].pivot_table(
        values='Revenue',
        index='Month',
        columns='Category',
        aggfunc='sum'
    )
    monthly_cat_trend.plot(marker='o')
    plt.title('Monthly Revenue Trend of Top Categories')
    plt.xlabel('Month')
    plt.ylabel('Revenue (₹)')
    plt.xticks(rotation=45)
    
    plt.tight_layout()
    plt.show()
    
    conn.close()
    return analysis

# Run the analysis
detailed_analysis = advanced_retail_analysis()

def advanced_retail_analysis():
    conn = sqlite3.connect('retail_analysis.db')
    
    # First, let's check our product categories
    print("Checking Product Categories:")
    categories = pd.read_sql_query("SELECT DISTINCT Category FROM products", conn)
    print(categories)
    
    # Basic analysis
    analysis = pd.read_sql_query("""
        SELECT 
            strftime('%m', s.Date) as Month,
            st.Location,
            p."Product Name",
            p.Category,
            COUNT(DISTINCT s."Transaction ID") as Num_Transactions,
            SUM(s.Quantity) as Units_Sold,
            ROUND(SUM(s."Total Amount"), 2) as Revenue
        FROM sales_transactions s
        JOIN products p ON s."Product ID" = p."Product ID"
        JOIN stores st ON s."Store ID" = st."Store ID"
        GROUP BY Month, st.Location, p.Category
        ORDER BY Month, Revenue DESC
    """, conn)
    
    print("\nSample of analysis data:")
    print(analysis.head())
    
    # Convert month numbers to names
    month_names = {
        '01': 'January', '02': 'February', '03': 'March', '04': 'April',
        '05': 'May', '06': 'June', '07': 'July', '08': 'August',
        '09': 'September', '10': 'October', '11': 'November', '12': 'December'
    }
    analysis['Month'] = analysis['Month'].map(month_names)
    
    # Print summary statistics
    print("\nRevenue by Category:")
    category_revenue = analysis.groupby('Category')['Revenue'].sum().sort_values(ascending=False)
    print(category_revenue)
    
    # Create visualization
    plt.figure(figsize=(12, 6))
    
    # Plot category performance
    category_revenue.plot(kind='bar')
    plt.title('Revenue by Category')
    plt.xlabel('Category')
    plt.ylabel('Total Revenue (₹)')
    plt.xticks(rotation=45)
    plt.tight_layout()
    plt.show()
    
    # Store performance by category
    plt.figure(figsize=(12, 6))
    store_cat_revenue = analysis.pivot_table(
        values='Revenue',
        index='Location',
        columns='Category',
        aggfunc='sum',
        fill_value=0
    )
    store_cat_revenue.plot(kind='bar', stacked=True)
    plt.title('Category Revenue by Store Location')
    plt.xlabel('Store Location')
    plt.ylabel('Revenue (₹)')
    plt.legend(title='Category', bbox_to_anchor=(1.05, 1))
    plt.tight_layout()
    plt.show()
    
    conn.close()
    return analysis

detailed_analysis = advanced_retail_analysis()
