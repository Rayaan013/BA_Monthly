# Import required libraries
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import sqlite3

# Read CSV files
sales_df = pd.read_csv('BA sales transaction.csv')
products_df = pd.read_csv('product sheet.csv')
stores_df = pd.read_csv('BA store info sheet.csv')

# Create database
conn = sqlite3.connect('retail_analysis.db')

# Create tables
sales_df.to_sql('sales_transactions', conn, if_exists='replace', index=False)
products_df.to_sql('products', conn, if_exists='replace', index=False)
stores_df.to_sql('stores', conn, if_exists='replace', index=False)

# Verify tables
cursor = conn.cursor()
print("Tables in database:")
cursor.execute("SELECT name FROM sqlite_master WHERE type='table';")
print(cursor.fetchall())

conn.close()

def analyze_store_performance():
    conn = sqlite3.connect('retail_analysis.db')
    
    store_sales = pd.read_sql_query("""
        SELECT 
            s."Store ID",
            st.Location,
            COUNT(*) as Number_of_Transactions,
            ROUND(SUM(s."Total Amount"), 2) as Total_Sales,
            ROUND(AVG(s."Total Amount"), 2) as Average_Transaction_Value
        FROM sales_transactions s
        JOIN stores st ON s."Store ID" = st."Store ID"
        GROUP BY s."Store ID"
        ORDER BY Total_Sales DESC
    """, conn)
    
    # Print store performance metrics
    print("\nStore Performance Analysis:")
    print("=" * 80)
    for _, row in store_sales.iterrows():
        print(f"\nStore Location: {row['Location']}")
        print(f"Number of Transactions: {row['Number_of_Transactions']}")
        print(f"Total Sales: {row['Total_Sales']:,.2f}")
        print(f"Average Transaction Value: {row['Average_Transaction_Value']:,.2f}")
        print("-" * 40)
    
    # Visualization
    plt.figure(figsize=(12, 6))
    plt.bar(store_sales['Location'], store_sales['Total_Sales'])
    plt.title('Sales by Store Location', fontsize=14, pad=20)
    plt.xlabel('Store Location', fontsize=12)
    plt.ylabel('Total Sales ()', fontsize=12)
    plt.xticks(rotation=45)
    plt.grid(True, linestyle='--', alpha=0.7)
    plt.tight_layout()
    plt.show()
    
    conn.close()
    return store_sales

# Run the analysis
store_performance = analyze_store_performance()
