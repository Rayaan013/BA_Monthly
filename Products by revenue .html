def analyze_product_profitability():
    conn = sqlite3.connect('retail_analysis.db')
    
    product_analysis = pd.read_sql_query("""
        SELECT 
            p."Product Name",
            COUNT(*) as Times_Sold,
            SUM(s.Quantity) as Total_Units,
            ROUND(SUM(s."Total Amount"), 2) as Total_Revenue,
            ROUND(AVG(s."Total Amount"), 2) as Avg_Transaction_Value
        FROM sales_transactions s
        JOIN products p ON s."Product ID" = p."Product ID"
        GROUP BY p."Product ID"
        ORDER BY Total_Revenue DESC
        LIMIT 10
    """, conn)
    
    # Print analysis
    print("\nTop 10 Products Analysis:")
    print("=" * 80)
    for _, row in product_analysis.iterrows():
        print(f"\nProduct: {row['Product Name']}")
        print(f"Times Sold: {row['Times_Sold']}")
        print(f"Total Units Sold: {row['Total_Units']}")
        print(f"Total Revenue: ₹{row['Total_Revenue']:,.2f}")
        print(f"Average Transaction Value: ₹{row['Avg_Transaction_Value']:,.2f}")
        print("-" * 40)
    
    # Visualization
    plt.figure(figsize=(12, 6))
    plt.bar(product_analysis['Product Name'], product_analysis['Total_Revenue'])
    plt.title('Top 10 Products by Revenue', fontsize=14, pad=20)
    plt.xlabel('Product Name', fontsize=12)
    plt.ylabel('Total Revenue (₹)', fontsize=12)
    plt.xticks(rotation=45)
    plt.grid(True, linestyle='--', alpha=0.7)
    plt.tight_layout()
    plt.show()
    
    conn.close()
    return product_analysis

# Run the analysis
product_performance = analyze_product_profitability()
